// This acts as your API to READ from the spreadsheet. // Deploy as Web App -> Execute as: Me -> Who has access: Anyone function doGet(e) { try { var requestedPackage = e.parameter.package; var action = e.parameter.action; var ss = SpreadsheetApp.getActiveSpreadsheet(); // NEW LOGIC: Fetch Submissions for the Database UI if (action === 'getSubmissions') { function getSubmissionsData(sheetName) { var sheet = ss.getSheetByName(sheetName); if (!sheet) return []; var data = sheet.getDataRange().getValues(); if (data.length <= 1) return []; // Return empty if only headers exist var headers = data.shift(); // Row 1 headers for submissions return data.map(function(row) { var obj = {}; headers.forEach(function(header, i) { if (header) { // Format timestamps cleanly for the dashboard if (Object.prototype.toString.call(row[i]) === '[object Date]') { obj[header.toString().trim()] = row[i].toLocaleDateString() + ' ' + row[i].toLocaleTimeString(); } else { obj[header.toString().trim()] = row[i]; } } }); return obj; }); } var payload = { unitPricing: getSubmissionsData("SUBMISSIONS_UNIT PRICING"), leadTimes: getSubmissionsData("SUBMISSIONS_LEAD TIMES"), priceEscalation: getSubmissionsData("SUBMISSIONS_PRICE ESCALATION") }; return ContentService.createTextOutput(JSON.stringify(payload)) .setMimeType(ContentService.MimeType.JSON); } // LOGIC: If no specific package is requested, return a list of unique packages for the landing page if (!requestedPackage) { var sheet = ss.getSheetByName("UNIT PRICING"); if (!sheet) throw new Error("UNIT PRICING tab not found."); var data = sheet.getDataRange().getValues(); data.shift(); // Skip Row 1 (The Note) var headers = data.shift(); // Row 2 (The Headers) var packageColIndex = headers.indexOf("BID PACKAGE"); if (packageColIndex === -1) throw new Error("Could not find 'BID PACKAGE' column."); var uniquePackages = []; var packageSet = new Set(); data.forEach(function(row) { var pkg = row[packageColIndex]; if (pkg && pkg.toString().trim() !== "") { packageSet.add(pkg.toString().trim()); } }); uniquePackages = Array.from(packageSet); return ContentService.createTextOutput(JSON.stringify({ packages: uniquePackages })) .setMimeType(ContentService.MimeType.JSON); } // LOGIC: If a package IS requested, fetch its specific items and notes function getFilteredData(sheetName) { var sheet = ss.getSheetByName(sheetName); if (!sheet) return { note: "", items: [] }; var data = sheet.getDataRange().getValues(); var noteRow = data.shift(); // Row 1 is now exclusively the Note var note = noteRow && noteRow[0] ? noteRow[0].toString().trim() : ""; var headers = data.shift(); // Row 2 is the actual Headers var formattedData = data.map(function(row) { var obj = {}; headers.forEach(function(header, i) { if (header) obj[header.toString().trim()] = row[i]; }); return obj; }); var filteredItems = formattedData.filter(function(row) { return row['BID PACKAGE'] === requestedPackage; }); return { note: note, items: filteredItems }; } // Return the payload with notes and items separated cleanly var payload = { bidPackage: requestedPackage, unitPricing: getFilteredData("UNIT PRICING"), leadTimes: getFilteredData("LEAD TIMES"), priceEscalation: getFilteredData("PRICE ESCALATION") }; return ContentService.createTextOutput(JSON.stringify(payload)) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({"error": error.toString()})) .setMimeType(ContentService.MimeType.JSON); } } // Required to handle complex CORS requests properly function doOptions(e) { return ContentService.createTextOutput("") .setMimeType(ContentService.MimeType.JSON); }
